    <label>Tenure (Months)</label>
    <input type="number" id="tenure" placeholder="Enter months" value="60">

    <label>Monthly EMI (₹)</label>
    <input type="number" id="emi" placeholder="Enter EMI amount" value="10624">

    <button onclick="calculateInterest()">Calculate</button>

    <div class="result" id="result" style="display:none;">
      <div>Total Payment <span id="totalPayment">₹0</span></div>
      <div>Total Interest Paid <span id="totalInterest">₹0</span></div>

  <script>
    function formatINR(num) {
      return num.toLocaleString('en-IN');
    }

    function calculateInterest() {
      const loanAmount = parseFloat(document.getElementById('loanAmount').value);
      const months = parseInt(document.getElementById('tenure').value);
      const emi = parseFloat(document.getElementById('emi').value);

      if (isNaN(loanAmount) || isNaN(months) || isNaN(emi) || loanAmount <= 0 || months <= 0 || emi <= 0) {
        alert("Please enter valid numbers for all fields.");
        return;
      }

      const totalPayment = emi * months;
      const totalInterest = totalPayment - loanAmount;

      document.getElementById('totalPayment').textContent = "₹" + formatINR(totalPayment.toFixed(0));
      document.getElementById('totalInterest').textContent = "₹" + formatINR(totalInterest.toFixed(0));
      document.getElementById('result').style.display = "block";
    }
  </script>
